Live Off Peak Usage = 
VAR SummaryTable =
    SUMMARIZE(
        KEEPFILTERS('fact_live_times'),
        'fact_live_times'[Hour],
        "AvgOcc", AVERAGE('fact_live_times'[PopularTimesLivePercent])
    )
VAR MinHourRow =
    TOPN(1, SummaryTable, [AvgOcc], ASC)
VAR HourNumber = MAXX(MinHourRow, [Hour])
VAR MinPercent = MAXX(MinHourRow, [AvgOcc])
VAR HourFormatted = FORMAT(TIME(HourNumber, 0, 0), "h AM/PM")

RETURN
FORMAT(COALESCE(MinPercent, 0), "0") & "% at " & HourFormatted
